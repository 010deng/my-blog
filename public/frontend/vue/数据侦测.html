# 对象的数据侦测

// 对象的数据侦测 getter 中收集依赖 setter 中通知依赖

// 首先需要一个管理依赖的类，具有增加依赖，删除依赖，通知依赖等功能。

// 使用 window.target 作为一个虚拟读取依赖的地方

class Dep {
    constructor() {
        this.subs = []
    }
    // 添加
    addSub(sub) {
        this.subs.push(sub)
    }

    removeSub(sub) {
        // 后面自定义的删除函数
        remove(this.subs, sub)
    }

    notify() {
        const subs = this.subs.slice()
        for (let i = 0; i < subs.length; i++) {
            subs[i].update()
        }
    }

    depend() {
        if (window.target) {
            this.addSub(window.target)
        }
    }
}

function remove(target, data) {
    if (target.length) {
        const idx = target.indexOf(data)
        // 存在
        if (idx > -1) {
            target.splice(idx, 1)
        }
    }
}

// 将数据变成响应式的

function defineReactive(data, key, val) {
    if (typeof val === 'object') {
        // 这里的修改接下面的递归操作
        new Observer(val)
    }
    // 方便管理依赖
    // 给每一个key都加上了一个数组来添加依赖
    let dep = new Dep()
    Object.defineProperty(data, key, {
        configurable: true,
        enumerable: true,
        get() {
            dep.depend()
            return val
        },
        set(newVal) {
            if (val === newVal) return
            val = newVal
            dep.notify()
        }
    })
}


// 现在需要一个能对数据做出响应的对象 ，数据发生变化的时候通知它，然后再由它通知其他地方

export class Watcher {
    constructor(vm, expOrFn, cb) {
        // expOrFn 是要解析的路径对象或者函数
        this.vm = vm
        // parsePath 为解析路径的函数，可以得到路径下的属性值
        this.getter = parsePath(expOrFn)
        this.cb = cb
        // 获取当前值
        this.value = this.get()
    }

    get() {
        // 将当前实例赋给window.target 实际上通过defineReactive() 将当前watcher加入了dep中
        window.target = this
        // 调用getter 函数，得到值
        let value = this.getter.call(this.vm, this.vm)
        window.target = undefined
        return value
    }
    // 数据改变触发更新
    update() {
        // 需要得到oldval，newval
        const oldVal = this.value
        let newVal = this.get()
        // 调用回调函数，第一个值为新值，后一个值为旧值
        this.cb.call(this.vm, newVal, oldVal)
    }
}

下面介绍如何解析某一个路径下的属性值


const baiRe = /[^\w.$]/

export function parsePath(path) {
    // . 结尾不符合路径要求
    if (baiRe.test(path)) return
    const segments = path.split('.')
    // 之后为核心递归操作
    return function (obj) {
        for (let i = 0; i < segments.length; i++) {
            if (!obj) return
            // 将每次读到的新值赋给obj，作为下一次的参数
            obj = obj[segments[i]]
        }
        // 返回的即为读取到的值
        return obj
    }

}

直到现在，我们已经实现了单个元素的数据侦测，然而对象往往都具有深层嵌套，所以考虑到这种情况。我们想到使用递归来解决这个问题。

// 封装一个observer类，作用是将所有元素包括其子元素都转换为响应式的

export class Observer {
    constructor(value) {
        this.value = value
        // 该方法只在对象时有用
        if (!Array.isArray(value)) {
            this.walk(value)
        }
    }

    walk(obj) {
        const keys = Object.keys(obj)
        for (let i = 0; i < keys.length; i++) {
            // 在 defineReactive 中加入了对象判断的逻辑，轻松实现递归，将其子元素也都响应化
            defineReactive(obj, keys[i], obj[keys[i]])
        }
    }

}

总结：对象数据侦测的关键点在于将数据变成响应式的，即getter收集依赖，setter触发依赖。而用户通过Watcher去读取数据的操作就会触发数据的getter，这样就会将这个Watcher主动加入到当前key的依赖数组Dep中。而修改数据的话，就将依赖项全部通知一遍触发更新。更新操作实际上收集了(newVal,oldVal)，并触发回调函数。

## bind()
> bind() 方法会创建一个新函数。当这个新函数被调用时，bind() 的第一个参数将作为它运行时的 this，之后的一序列参数将会在传递的实参前传入作为它的参数。

`bind()`的特性有以下几点
* 返回一个函数
* 可以传入参数
***注意这里的传参可以分步传参,即如下形式***

```js
var foo = {
    value: 1
};

function bar(name, age) {
    console.log(this.value);
    console.log(name);
    console.log(age);

}

var bindFoo = bar.bind(foo, 'daisy');
bindFoo('18');
// 1
// daisy
// 18
```
* 如果绑定函数作为构造函数使用，`即其中使用了new`，那么传入的this失效，但是参数有效

下面基于bind的特征来大致实现bind的效果

```js
Function.prototype.myBind() = function(context) {
        if (typeof this !== 'function') {
            // 必须通过函数来调用
            throw TypeError("Bind must be called on a function");
        }
        var that = this
        // 获取从第二个参数开始的所有参数，第一个参数是this
        const args = Array.prototype.slice.call(arguments,1)
        // 定义一个空函数作为后面函数原型使用，目的是避免直接赋值的原型污染
        var NOP = function() {}
        var Bound = function() {
            // 获取所有绑定函数的参数
            var bindArgs = Array.prototype.slice.call(arguments)
            // 这里是判断是否是作为构造函数来使用，改变this的指向，后面使用的this为改变后的this
            return that.apply(this instanceof NOP ? this : context , args.concat(bindArgs))
        }
        // 让其可以使用原型上的值，并且避免了直接改变Bound原型
        NOP.prototype = this.prototype
        Bound.prototype = new NOP()
        return Bound
}
```

下面贴一个绑定函数作为构造函数的情景，就可以说明改变原型是很有必要的
```js
var value = 2;

var foo = {
    value: 1
};

function bar(name, age) {
    this.habit = 'shopping';
    console.log(this.value);
    console.log(name);
    console.log(age);
}

bar.prototype.friend = 'kevin';

var bindFoo = bar.bind(foo, 'daisy');

var obj = new bindFoo('18');
// undefined
// daisy
// 18
console.log(obj.habit);
console.log(obj.friend);
// shopping
// kevin

```

## call,apply
> call和apply的实现原理差不多，两者的区别就在于call是传递参数列表，而apply是传一个数组

下面将分成两个版本来说明

不使用ES6
```js
// 基本思路是将调用call的函数加入到设置this对象的属性中，调用以后再删除

Function.prototype.myCall = function(context) {
    // 当传入为null的时候，默认为window
    var context = context || window
    // fn 是自己定义的属性名，this 是获取是谁调用了 call
    context.fn = this
    var args = []
    // 从第二位开始取参数
    for (let i = 1; i < arguments.length; i++) {
        args.push('arguments[' + i + ']')
    }
    // eval 传入字符串，相当于script标签执行其中的语句
    // 调用fn属性（this正确），并传入args参数列表
    // args自动执行toString方法
    var ret = eval('context.fn('+ args +')')
    delete context.fn
    return ret
}



// apply
Function.prototype.apply = function (context, arr) {
    var context = Object(context) || window
    context.fn = this
    var ret
    // 如果没传入后面的参数数组
    if (!arr) {
        ret = context.fn()
    } else {
        var args = []
        // 这里直接循环arr，所以是从0开始
        for (var i = 0, len = arr.length; i < len; i++) {
            args.push('arr[' + i + ']')
        }
        ret = eval('context.fn(' + args + ')')
    }
    delete context.fn
    return ret
}

```
使用ES6

```js
// call
Function.prototype.myCall = function (context) {
    if (typeof this !== 'function') {
        throw new Error('error')
    }
    // 防止命名冲突
    const fn =  Symbol('fn')
    var context = context || window
    context[fn] = this

    // 从第二个参数取起
    const args = [...arguments].slice(1)
    let ret = context[fn](...args)
    delete context[fn]
    return ret
}


// apply
Function.prototype.myApply = function (context) {
    if (typeof this !== 'function') {
        throw this + ' is not a function';
    }
    const args = arguments[1]
    const fn = Symbol('fn')
    context[fn] = this
    const ret = context[fn](...args)
    delete context[fn]
    return ret
}

```
